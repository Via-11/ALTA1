DROP TABLE IF EXISTS documents;
CREATE TABLE documents (
    document_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    document_type VARCHAR(100) NOT NULL COMMENT 'Type: resume, id, certificate, business_plan, pitch_deck, other',
    file_name VARCHAR(255) NOT NULL COMMENT 'Original file name',
    file_path VARCHAR(500) NOT NULL COMMENT 'Path to stored file',
    file_size INT NULL COMMENT 'File size in bytes',
    file_type VARCHAR(100) NULL COMMENT 'MIME type',
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT 'Upload timestamp',
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT 'Last update timestamp',
    
    CONSTRAINT fk_documents_user 
        FOREIGN KEY (user_id) 
        REFERENCES users(user_id) 
        ON DELETE CASCADE,
    
    INDEX idx_user_id (user_id),
    INDEX idx_document_type (document_type),
    INDEX idx_uploaded_at (uploaded_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
COMMENT='Stores user uploaded documents';

DROP VIEW IF EXISTS user_profile_summary;
CREATE VIEW user_profile_summary AS
SELECT 
    u.user_id,
    u.name,
    u.email,
    u.role,
    u.company,
    u.contact,
    u.bio,
    u.position,
    u.skills,
    u.github,
    u.linkedin,
    u.facebook,
    u.website,
    COUNT(d.document_id) AS total_documents,
    u.created_at,
    u.updated_at
FROM users u
LEFT JOIN documents d ON u.user_id = d.user_id
GROUP BY u.user_id;

-- View to get active document uploads
DROP VIEW IF EXISTS recent_documents;
CREATE VIEW recent_documents AS
SELECT 
    d.*,
    u.name AS user_name,
    u.email AS user_email
FROM documents d
JOIN users u ON d.user_id = u.user_id
ORDER BY d.uploaded_at DESC;

DROP PROCEDURE IF EXISTS GetUserProfile;
DELIMITER $$

CREATE PROCEDURE GetUserProfile(IN p_user_id INT)
BEGIN
    SELECT * FROM user_profile_summary WHERE user_id = p_user_id;
END$$

DELIMITER ;

-- Procedure to get user documents
DROP PROCEDURE IF EXISTS GetUserDocuments;
DELIMITER $$

CREATE PROCEDURE GetUserDocuments(IN p_user_id INT)
BEGIN
    SELECT * FROM documents 
    WHERE user_id = p_user_id 
    ORDER BY uploaded_at DESC;
END$$

DELIMITER ;

DROP PROCEDURE IF EXISTS LogProfileActivity;

CREATE PROCEDURE LogProfileActivity(
    IN p_user_id INT,
    IN p_action VARCHAR(100),
    IN p_field_name VARCHAR(100),
    IN p_old_value TEXT,
    IN p_new_value TEXT,
    IN p_ip_address VARCHAR(45)

BEGIN
    INSERT INTO profile_activity_log 
    (user_id, action, field_name, old_value, new_value, ip_address)
    VALUES 
    (p_user_id, p_action, p_field_name, p_old_value, p_new_value, p_ip_address);
END$$
